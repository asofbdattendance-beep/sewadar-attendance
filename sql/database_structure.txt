CREATE TABLE public.attendance (
  id bigint NOT NULL DEFAULT nextval('attendance_id_seq'::regclass),
  badge_number text NOT NULL,
  sewadar_name text NOT NULL,
  centre text NOT NULL,
  department text,
  type text NOT NULL CHECK (type = ANY (ARRAY['IN'::text, 'OUT'::text])),
  scan_time timestamp with time zone DEFAULT now(),
  scanner_badge text NOT NULL,
  scanner_name text NOT NULL,
  scanner_centre text NOT NULL,
  latitude double precision,
  longitude double precision,
  device_id text,
  created_at timestamp with time zone DEFAULT now(),
  session_id bigint,
  manual_entry boolean NOT NULL DEFAULT false,
  submitted_by text NOT NULL,
  submitted_at timestamp with time zone NOT NULL DEFAULT now(),
  duty_type text NOT NULL DEFAULT 'gate_entry'::text CHECK (duty_type = ANY (ARRAY['satsang'::text, 'gate_entry'::text, 'watch_ward'::text])),
  CONSTRAINT attendance_pkey PRIMARY KEY (id),
  CONSTRAINT attendance_badge_number_fkey FOREIGN KEY (badge_number) REFERENCES public.sewadars(badge_number),
  CONSTRAINT fk_attendance_session_id FOREIGN KEY (session_id) REFERENCES public.attendance_sessions(id)
);
